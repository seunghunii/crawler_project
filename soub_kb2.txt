		function chngYear() {
			var setMon = ""; 
			var tmp = $("#기준월").val();
			$("#기준월").empty();
			var date = new Date();
			var y = date.getFullYear();

				maxMon = m;

					setMon = m;
					if (m < 10) {
						setMon = "0"+m;
					}
				} else {
					setMon  = tmp;
				}
			} else {
				setMon  = tmp;
			}
			for (var i = 1; i <= maxMon ; i++) {
				if (i < 10) {
					$("#기준월").append("<option value='0"+i+"'>"+i+"월</option>");
				}else {
					$("#기준월").append("<option value='"+i+"'>"+i+"월</option>");
				}
			}
			$("#기준월").val(setMon).prop("selected", true);
		}
